Q2. Create a View Emp_Salary_Details which will display employee’s name, salary,
commission, total (which will be salary plus commission), tax as 30% of the total,
HRA as 10% of salary, DA as 5% of salary, TA as 7% of salary, PF as 200 and Gross
that would be the formula as (Total-Tax) + (HRA + DA + TA) – PF

Query: CREATE VIEW Emp_Salary_Details AS
    -> SELECT
    ->     ENAME AS "Employee Name",
    ->     SAL AS "Salary",
    ->     COMM AS "Commission",
    ->     (SAL + IFNULL(COMM,0)) AS "Total",
    ->     (0.30 * (SAL + IFNULL(COMM,0))) AS "Tax",
    ->     (0.10 * SAL) AS "HRA",
    ->     (0.05 * SAL) AS "DA",
    ->     (0.07 * SAL) AS "TA",
    ->     200 AS "PF",
    ->     ((SAL + IFNULL(COMM,0)) - (0.30 * (SAL + IFNULL(COMM,0)))
    ->         + ((0.10 * SAL) + (0.05 * SAL) + (0.07 * SAL))
    ->         - 200) AS "Gross"
    -> FROM emp;

SELECT * FROM Emp_Salary_Details;

Q4. Create a view named Dept_High_Earners which will display employee name, job,

Query: CREATE VIEW Dept_High_Earners AS

    e.ENAME AS "Employee Name",
    e.JOB   AS "Job",
    e.SAL   AS "Salary",
    d.DNAME AS "Department Name"
FROM emp e

Q1. Produce the following output 

Query: SELECT 
    e.ENAME AS "Employee Name",
    e.JOB   AS "Job",
    e.SAL   AS "Salary",
    j.MaxSal,
    j.MinSal,
    j.AvgSal
FROM emp e
JOIN (
    SELECT JOB, MAX(SAL) AS MaxSal, MIN(SAL) AS MinSal, AVG(SAL) AS AvgSal
    FROM emp
    GROUP BY JOB
) j
ON e.JOB = j.JOB;

Q2. Produce the following output 
Query: SELECT 
    e.ENAME AS "Employee Name",
    e.DEPTNO AS "Dept No",
    e.SAL AS "Existing Salary",
    d_stats.MaxSal AS "Maximum Salary (Dept)",
    d_stats.MinSal AS "Minimum Salary (Dept)",
    d_stats.AvgSal AS "Average Salary (Dept)",
    (e.SAL - d_stats.AvgSal) AS "Dept Difference"
FROM emp e
JOIN (
    SELECT 
        DEPTNO,
        MAX(SAL) AS MaxSal,
        MIN(SAL) AS MinSal,
        AVG(SAL) AS AvgSal
    FROM emp
    GROUP BY DEPTNO
) d_stats
ON e.DEPTNO = d_stats.DEPTNO;
