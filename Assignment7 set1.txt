1) Create a trigger on MyEmp2 table which will transfer the deleted records from
myEmp2 table into another table Backup_MyEmp2.
Note  Backup_MyEmp2 table’s structure should be same as that of MyEmp2 table

CREATE TABLE BACKUP_MYEMP2(
	BACKUP_ID INT PRIMARY KEY AUTO_INCREMENT,
	EMPLOYEE_ID INT,
	ACTION VARCHAR(20),
	ACTION_TIME TIMESTAMP DEFAULT CURRENT_TIMESTAMP);




DELIMITER $$
	CREATE TRIGGER after_employee_DELETE
	AFTER DELETE ON MY_EMP2
	FOR EACH ROW
	BEGIN 
		INSERT INTO BACKUP_MYEMP2(EMPLOYEE_ID,ACTION) VALUES(OLD.EMPNO,'DELETE');
	END $$
DELIMITER ;

5) Create a table MyEmp4 which will be copy of MyEmp2 along with all its records.
Create a table Employees_Total_Salary which will have empno and total salary as
the two columns.
Whenever the salary gets updated from MyEmp4 table then the updated salary +
commission should get added in the Employees_Total_Salary table. For employees
who do not have any commission consider it as 10% of salary.

CREATE OR REPLACE TRIGGER trg_total_salary
AFTER UPDATE OF sal ON MyEmp4
FOR EACH ROW
DECLARE
    v_comm NUMBER;
    v_total NUMBER;
BEGIN
    IF :NEW.comm IS NULL THEN
        v_comm := :NEW.sal * 0.10;
    ELSE
        v_comm := :NEW.comm;
    END IF;
    v_total := :NEW.sal + v_comm;

    MERGE INTO Employees_Total_Salary t
    USING (SELECT :NEW.empno AS empno FROM dual) s
    ON (t.empno = s.empno)
    WHEN MATCHED THEN
        UPDATE SET t.total_salary = v_total
    WHEN NOT MATCHED THEN
        INSERT (empno, total_salary)
        VALUES (:NEW.empno, v_total);
END;
