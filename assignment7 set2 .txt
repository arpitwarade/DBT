    -> BEGIN
    ->     -- Declare variables
    ->     DECLARE v_ename VARCHAR(50);
    ->     DECLARE v_sal DECIMAL(10,2);
    ->     DECLARE done INT DEFAULT 0;
    ->
    ->     -- Declare cursor
    ->     DECLARE emp_cursor CURSOR FOR
    ->         SELECT ename, sal FROM emp;
    ->
    ->     -- Declare handler for end of cursor
    ->     DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = 1;
    ->
    ->     -- Open cursor
    ->     OPEN emp_cursor;
    ->

    ->         FETCH emp_cursor INTO v_ename, v_sal;

    ->             LEAVE read_loop;
    ->         END IF;

    ->         -- Display employee name and salary
    ->         SELECT CONCAT('Employee: ', v_ename, ' | Salary: ', v_sal) AS Output;

Q8. Create a procedure PC8 that uses a cursor to print employee names along with experience in years from HIREDATE till today.

    -> BEGIN
    ->     -- Variables
    ->     DECLARE v_ename VARCHAR(50);
    ->     DECLARE v_hiredate DATE;
    ->     DECLARE v_experience INT;
    ->     DECLARE done INT DEFAULT 0;
    ->
    ->     -- Cursor declaration
    ->     DECLARE emp_cursor CURSOR FOR
    ->         SELECT ename, hiredate FROM emp;
    ->
    ->     -- Handler for end of cursor
    ->     DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = 1;
    ->
    ->     -- Open cursor
    ->     OPEN emp_cursor;
    ->
    ->     -- Loop through rows
    ->     read_loop: LOOP
    ->         FETCH emp_cursor INTO v_ename, v_hiredate;
    ->         IF done = 1 THEN
    ->             LEAVE read_loop;
    ->         END IF;
    ->
    ->         -- Calculate experience in years
    ->         SET v_experience = TIMESTAMPDIFF(YEAR, v_hiredate, CURDATE());
    ->
    ->         -- Display employee name and experience
    ->         SELECT CONCAT('Employee: ', v_ename, ' | Experience: ', v_experience, ' years') AS Output;
    ->     END LOOP;
    ->
    ->     -- Close cursor
    ->     CLOSE emp_cursor;
    -> END$$
