1 Display the oldest employee in the deptno 30.
 
Query : 
        mysql> SELECT * FROM emp
    -> WHERE deptno = 30
    -> ORDER BY hiredate ASC
    -> LIMIT 1;

job SALESMAN. The records of SALESMAN should be excluded from the
Report.

        select * from emp
    -> where sal<(select AVG(sal) from emp where job = 'SALESMAN')

6 Display the name, hire date, and salary for all employees who have both the same
salary and commission as Scott.
Note: Do not display SCOTT in the result set.

Query : 
	mysql> select ename, hiredate, sal from emp where (sal = (select sal from emp where ename = 'SCOTT') or comm = (select comm from emp where ename = 'SCOTT')) AND ename <> "SCOTT";
	

7 Display the department names of employees who have their salary greater than
Martin’s salary.

Query :
        mysql> select d.dname,e.ename
    -> from emp e join dept d
    -> on e.deptno=d.deptno
    -> where e.sal>(select sal from emp where ename='Martin');
	

8 Display the department name that has the highest salary record.

Query : 
        mysql> select d.dname,max(e.sal)
    -> from emp e join dept d
    -> on e.deptno=d.deptno
    -> group by d.deptno order by max(e.sal) desc limit 1;
