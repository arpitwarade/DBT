1) Create a procedure P1 that inserts numbers from 1 to 10 into table t1 which has only
one column named a of int type. Use while loop only.

DELIMITER $$

CREATE PROCEDURE P1()

BEGIN

	DECLARE NUM INT DEFAULT 1;

	WHILE NUM <= 10

	DO
	insert into t1 values(NUM);
	SELECT NUM;
	SET NUM = NUM + 1;

	END WHILE;	

END $$

DELIMITER ;



2) Create a procedure P2 that inserts only even numbers between 1 and 50 into table t2
which has only one column named b of int type. Use repeat loop only.

DELIMITER $$

CREATE PROCEDURE P3()

BEGIN
      
   DECLARE b INT DEFAULT 2;

   REPEAT 
        insert into t2 values(b);
        SET b = b + 2;
        SELECT b ;
        UNTIL b > 50 
   END REPEAT;

END$$

3) Create a procedure P3 that calculates sum of numbers from 1 to 100 and inserts the
final sum as a single row in table t3 which has only one column named c of int type.
Use of while loop only.

DELIMITER $$

	CREATE PROCEDURE P3()

	BEGIN

		DECLARE NUM INT DEFAULT 1;
		DECLARE SUM INT DEFAULT 0;

		WHILE NUM <= 100 DO

			SET SUM = SUM + NUM;
			SET NUM = NUM + 1;

		END WHILE;
		INSERT INTO T3 VALUES(SUM);

	END $$

DELIMITER ;

6) Create a procedure P6 that accepts marks as an input parameter and determines the
student’s result. If the marks are greater than or equal to 80, the procedure should
display the message 'Distinction'; if the marks are greater than or equal to 60 but less
than 80, it should display 'First Class'; if the marks are greater than or equal to 40 but
less than 60, it should display 'Pass'; otherwise, it should display 'Fail'.

DELIMITER $$
	CREATE PROCEDURE P6(IN MARKS INT)
	BEGIN
		IF MARKS >= 80 THEN
			SELECT 'DISTINCTION' AS RESULT;
		ELSEIF MARKS >= 60 THEN 
			SELECT 'FIRST CLASS' AS RESULT;
		ELSEIF MARKS >= 40 THEN
			SELECT 'PASS' AS RESULT;
		ELSE 
			SELECT 'FAIL' AS RESULT;
		END IF; 	
	END $$
DELIMITER ;

8) Create a procedure P8 that displays numbers starting from 1. Stop loop when
number becomes divisible by 7. Display numbers until that point only. Use basic loop
only.

DELIMITER $$

CREATE PROCEDURE P8()
BEGIN
    DECLARE num INT DEFAULT 1;

    basic_loop: LOOP
        -- Display the current number
        SELECT num AS Number;

        -- Exit condition: stop when divisible by 7
        IF num % 7 = 0 THEN
            LEAVE basic_loop;
        END IF;

        -- Increment the counter
        SET num = num + 1;
    END LOOP basic_loop;
END $$

DELIMITER ;
